Code Style

Example structure

public sealed class VehicleDriveController : MonoBehaviour
{
    [SerializeField] private float acceleration = 8f;
    [SerializeField] private float turnSpeed = 90f;

    public void Tick(float throttle, float steering)
    {
        transform.position += transform.forward * throttle * acceleration * Time.deltaTime;
        transform.Rotate(Vector3.up, steering * turnSpeed * Time.deltaTime);
    }
}
